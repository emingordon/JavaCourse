import java.util.List;

public class ElementFind {

    public static void main(String[] args) {
        List<String> words = List.of("fight", "grape", "foster", "fight");
        System.out.println(findIndex(words, "fight"));
        System.out.println(findIndex(words, "foster"));
        System.out.println(findIndex(words, "top"));

        List<Integer> numbers = List.of(1, 2, 3, 4);
        System.out.println(findIndex(numbers, 2));
        System.out.println(findIndex(numbers, 4));
    }

    public static <T> int findIndex(List<T> list, T target) {
        return list.indexOf(target);
    }
}
