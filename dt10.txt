import java.time.DayOfWeek;
import java.time.LocalDate;

public class Fridays {

    public static void main(String[] args) {
        LocalDate nextFriday = LocalDate.now().plusDays((DayOfWeek.FRIDAY.getValue() - LocalDate.now().getDayOfWeek().getValue() + 7) % 7);
        LocalDate previousFriday = LocalDate.now().minusDays((LocalDate.now().getDayOfWeek().getValue() - DayOfWeek.FRIDAY.getValue() + 7) % 7);

        System.out.println("Next Friday: " + nextFriday);
        System.out.println("Previous Friday: " + previousFriday);
    }
}
