import java.time.LocalDate;
import java.time.temporal.TemporalAdjusters;
import java.time.temporal.WeekFields;
import java.util.Locale;

public class MinCalendar {

    public static void main(String[] args) {
        LocalDate currentDate = LocalDate.now();


        LocalDate firstDayOfYear = currentDate.with(TemporalAdjusters.firstDayOfYear());
        Integer minDayOfYear = firstDayOfYear.getDayOfYear();

        LocalDate firstDayOfMonth = currentDate.with(TemporalAdjusters.firstDayOfMonth());
        Integer minDayOfMonth = firstDayOfMonth.getDayOfMonth();

        WeekFields weekFields = WeekFields.of(Locale.getDefault());
        Integer minWeekOfYear = currentDate.with(TemporalAdjusters.firstDayOfYear()).get(weekFields.weekOfYear());

        LocalDate firstDayOfWeek = currentDate.with(TemporalAdjusters.previousOrSame(weekFields.getFirstDayOfWeek()));
        Integer minDayOfWeek = firstDayOfWeek.getDayOfWeek().getValue();

        System.out.println("Current Date: " + currentDate);
        System.out.println("Minimum day of the year: " + minDayOfYear);
        System.out.println("Minimum day of the month: " + minDayOfMonth);
        System.out.println("Minimum week of the year: " + minWeekOfYear);
        System.out.println("Minimum day of the week: " + minDayOfWeek);
    }
}
