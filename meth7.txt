public class meth7 {
    public static void main(String[] args) {
        String text1 = "Sinister";
        String text2 = "Correct";
        String text3 = "Good";

        System.out.println(validateText(text1));
        System.out.println(validateText(text2));
        System.out.println(validateText(text3));
    }

    public static String validateText(String text) {
        if (text.length() < 3 || text.length() > 8) {
            return "Error: Wrong text";
        } else {
            return "Okay";
        }
    }
}
