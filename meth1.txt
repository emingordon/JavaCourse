public class meth1 {

    public static int getTotal(int a, int b) {
        return a + b;
    }

    public static void main(String[] args) {
        int a = 10;
        int b = 8;

        int sum = getTotal(a, b);
        System.out.println("The sum of " + a + " and " + b + " is: " + sum);
    }
}
