import java.util.ArrayList;
import java.util.List;

public class MultiplyArrays {

    public static void main(String[] args) {
        ArrayList<Double> numbers = new ArrayList<>(List.of(3.0, 5.0, 7.0));
        double multiplier = 2.0;
        ArrayList<Double> result = manyList(numbers, multiplier);
        System.out.println(result);
    }

    public static ArrayList<Double> manyList(ArrayList<Double> list, double multiplier) {
        ArrayList<Double> resultList = new ArrayList<>();
        list.forEach(number -> resultList.add(number * multiplier));
        return resultList;
    }
}
