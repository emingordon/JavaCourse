import java.util.ArrayList;
import java.util.List;

public class LastElement {

    public static void main(String[] args) {
        ArrayList<String> strings = new ArrayList<>(List.of("black", "white", "dark"));
        System.out.println(getLastElement(strings));

        ArrayList<String> emptyList = new ArrayList<>(List.of());
        System.out.println(getLastElement(emptyList));
    }

    public static String getLastElement(ArrayList<String> list) {
        return list.isEmpty() ? "" : list.get(list.size() - 1);
    }
}
