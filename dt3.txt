import java.time.LocalDate;
import java.time.temporal.TemporalAdjusters;
import java.time.temporal.WeekFields;
import java.util.Locale;

public class MaxCalendar {

    public static void main(String[] args) {
        LocalDate currentDate = LocalDate.now();

        LocalDate lastDayOfYear = currentDate.with(TemporalAdjusters.lastDayOfYear());
        Integer maxDayOfYear = lastDayOfYear.getDayOfYear();

        LocalDate lastDayOfMonth = currentDate.with(TemporalAdjusters.lastDayOfMonth());
        Integer maxDayOfMonth = lastDayOfMonth.getDayOfMonth();

        WeekFields weekFields = WeekFields.of(Locale.getDefault());
        Integer maxWeekOfYear = currentDate.with(TemporalAdjusters.lastDayOfYear()).get(weekFields.weekOfYear());

        LocalDate lastDayOfWeek = currentDate.with(TemporalAdjusters.nextOrSame(weekFields.getFirstDayOfWeek().minus(1)));
        Integer maxDayOfWeek = lastDayOfWeek.getDayOfWeek().getValue();

        System.out.println("Current Date: " + currentDate);
        System.out.println("Maximum day of the year: " + maxDayOfYear);
        System.out.println("Maximum day of the month: " + maxDayOfMonth);
        System.out.println("Maximum week of the year: " + maxWeekOfYear);
        System.out.println("Maximum day of the week: " + maxDayOfWeek);
    }
}
